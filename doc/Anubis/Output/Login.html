def message
  case self.result
  when -1
    return I18n.t('errors.invalid_login_parameters')
  when -2
    return I18n.t('errors.invalid_login_or_password')
  else
    return super
  end
end

def to_h
  result = super.to_h
  return result if self.result != 0
  result.merge!({
                    token: self.token,
                    name: self.name,
                    surname: self.surname,
                    email: self.email,
                    locale: self.locale
                })
  result
end
