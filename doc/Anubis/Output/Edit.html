def to_h
  result = super.to_h
  return result if self.result != 0
  result[:title] = self.title if self.title != ''
  result[:fields] = self.fields if self.fields
  result[:action] = self.action if self.action != ''
  result.merge!({
                    values: self.values,
                    options: self.options_to_json(self.options)
                })
  result
end
