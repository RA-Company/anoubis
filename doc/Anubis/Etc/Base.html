def initialize(options = {})
  self.data = nil
  self.menu = nil
  self.tab = nil
  self.action = ''

  if options.key? :params
    self.action = options[:params][:action] if options[:params].key? :action
    if options[:params].key? :time
      self.time = options[:params][:time].to_s.to_i
    else
      self.time = 0
    end
  end
end
